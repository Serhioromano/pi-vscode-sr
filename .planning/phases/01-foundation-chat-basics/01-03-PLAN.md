    Instead of `export let` (which creates ESM readonly-binding issues and violates D-02), use a single `export const state` object whose fields are mutable:
    
    ```typescript

    **Why this pattern:**
    - D-02 says no module-level mutable `let`/`const` — `const state` with mutable object fields satisfies this (the binding is const, the contents mutate via object property assignment)

    import type { ExtensionAPI } from "@earendil-works/pi-coding-agent";
    import { registerWriteOverride } from "./tool-overrides.js";
    import { registerEditOverride } from "./tool-overrides.js";
    import {
      state,
      isVscodeReady,
      cleanupPiDir,
    } from "./review-lifecycle.js";

    export default function (pi: ExtensionAPI) {
      pi.on("session_start", () => {
        state.sessionReviewIds.clear();
        state.sessionApproveAll.clear();
        state.vscodeNotOpenWarned = false;
        const cwd = process.cwd();
        if (!isVscodeReady(cwd)) {
          console.warn(
            "VS Code not detected — working without diff review. " +
            "All file changes will be applied directly. " +
            "Open this project in VS Code with Serhioromano.vscode-pi-sr extension " +
            "installed to enable visual review."
          );
          state.vscodeNotOpenWarned = true;
        }
      });

      pi.on("before_agent_start", () => {
        state.sessionApproveAll.clear();
      });

      pi.on("message_end", () => {
        cleanupPiDir();
      });

      registerWriteOverride(pi);
      registerEditOverride(pi);
    }
    ```
